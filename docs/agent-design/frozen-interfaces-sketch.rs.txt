// ============================================================================
pub mod decision {
    use browser_use_llm::schema::FinishReason;

    // ---- loop_decision.rs (turn.rs:168-355, 677) ----
    #[derive(Debug, Clone, Default)]
    pub struct SamplingOutcome {
        pub model_needs_follow_up: bool,            // turn.rs:250
        pub last_agent_message: Option<String>,
        pub finish_reason: Option<FinishReason>,
    }
    #[derive(Debug, Clone)]
    pub struct TokenStatus {
        pub auto_compact_scope_tokens: i64,
        pub auto_compact_scope_limit: i64,
        pub full_context_window_limit_reached: bool,
        pub token_limit_reached: bool,              // scope>=limit || full_window (turn.rs:677-678)
    }
    pub fn needs_follow_up(model_nfu: bool, has_pending_input: bool) -> bool { model_nfu || has_pending_input } // turn.rs:255
    pub fn token_limit_reached(scope: i64, limit: i64, full: bool) -> bool { scope >= limit || full }           // turn.rs:677
    pub fn should_compact_mid_turn(tlr: bool, nfu: bool) -> bool { tlr && nfu }                                 // turn.rs:282
    pub fn can_drain_after_compact(model_nfu: bool) -> bool { !model_nfu }                                       // turn.rs:306
    pub fn initial_can_drain(turn_has_fresh_input: bool) -> bool { !turn_has_fresh_input }                       // turn.rs:168

    #[derive(Debug, Clone, PartialEq, Eq)]
    pub enum LoopStep { CompactThenContinue { can_drain_next: bool }, Continue, Complete }
    pub fn classify_loop_step(out: &SamplingOutcome, has_pending_input: bool, st: &TokenStatus) -> LoopStep {
        let nfu = needs_follow_up(out.model_needs_follow_up, has_pending_input);
        if should_compact_mid_turn(st.token_limit_reached, nfu) {
            LoopStep::CompactThenContinue { can_drain_next: can_drain_after_compact(out.model_needs_follow_up) }
        } else if nfu { LoopStep::Continue } else { LoopStep::Complete }
    }

    // ---- retry.rs (turn.rs:924-1021) ----
    #[derive(Debug, Clone, PartialEq, Eq)]
    pub enum RetryAction { Fail, SwitchTransport, Backoff { delay_ms: u64 } }
    pub fn retry_decision(retries: u32, max: u32, retryable: bool, can_switch_transport: bool,
                          requested_delay_ms: Option<u64>) -> RetryAction {
        if !retryable { return RetryAction::Fail; }
        if retries >= max { return if can_switch_transport { RetryAction::SwitchTransport } else { RetryAction::Fail }; }
        RetryAction::Backoff { delay_ms: requested_delay_ms.unwrap_or(backoff_ms(retries)) }
    }
    pub fn backoff_ms(retries: u32) -> u64 { 200u64.saturating_mul(1u64 << retries.min(6)) } // deterministic, pure

    // ---- tool_decision.rs (sandboxing.rs:202-334 + orchestrator.rs:145-388) ----
    #[derive(Debug, Clone, Copy, PartialEq, Eq)]
    pub enum ToolParallelism { Parallel, Serial }   // Parallel=RwLock read, Serial=RwLock write
    pub fn classify_parallelism(parallel_safe: bool, model_supports: bool) -> ToolParallelism {
        if parallel_safe && model_supports { ToolParallelism::Parallel } else { ToolParallelism::Serial }
    }
    // default_exec_approval_requirement / sandbox_override_for_first_attempt / plan_attempts / map_decision
    // are declared in the tools:: module (they reference its enums) but stay PURE — see tools below.
}
